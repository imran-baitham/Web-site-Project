JavaScript Code


Select input id using getElementById() to create Auto Text Effect using javascript and The onclick() method is used to handle the click event for element.


                  
     const textEl = document.getElementById("text");
     const speedEl = document.getElementById("speed");
     const text = "We Love Programming!";
     let idx = 1;
     let speed = 300;

     function writeText() {
          textEl.innerText = text.slice(0, idx);
          idx++;
          if (idx > text.length) {
               idx = 1;
          }
          console.log(idx);
          setTimeout(writeText, speed);
          speedEl.addEventListener("input", (e) => (speed = 300 / e.target.value));
     }
     writeText();
